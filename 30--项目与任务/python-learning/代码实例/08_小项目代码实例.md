# 小项目代码实例

## 项目 1：命令行 Todo

```python
import json
from pathlib import Path

TODO_FILE = Path("todos.json")


def load_todos() -> list[dict]:
    if not TODO_FILE.exists():
        return []
    return json.loads(TODO_FILE.read_text(encoding="utf-8"))


def save_todos(todos: list[dict]):
    TODO_FILE.write_text(
        json.dumps(todos, ensure_ascii=False, indent=2),
        encoding="utf-8",
    )


def add_todo(title: str):
    todos = load_todos()
    todos.append({"title": title, "done": False})
    save_todos(todos)


def list_todos():
    todos = load_todos()
    for index, todo in enumerate(todos, start=1):
        status = "完成" if todo["done"] else "未完成"
        print(f"{index}. [{status}] {todo['title']}")


add_todo("学习函数")
add_todo("完成作业")
list_todos()
```

## 项目 2：文件整理器

```python
from pathlib import Path
import shutil


def organize(folder: str):
    root = Path(folder)

    for path in root.iterdir():
        if not path.is_file():
            continue

        suffix = path.suffix.lower().lstrip(".") or "no_extension"
        target_dir = root / suffix
        target_dir.mkdir(exist_ok=True)

        shutil.move(str(path), str(target_dir / path.name))


organize("downloads")
```

## 项目 3：学生成绩管理

```python
students = []


def add_student(name: str, score: float):
    students.append({"name": name, "score": score})


def average_score() -> float:
    if not students:
        return 0
    return sum(item["score"] for item in students) / len(students)


def show_students():
    for student in students:
        print(student["name"], student["score"])


add_student("Alice", 90)
add_student("Bob", 80)
show_students()
print("平均分：", average_score())
```

## 项目 4：简易记账本

```python
records = []


def add_record(category: str, amount: float, note: str):
    records.append({
        "category": category,
        "amount": amount,
        "note": note,
    })


def total_amount() -> float:
    return sum(record["amount"] for record in records)


add_record("餐饮", 35, "午饭")
add_record("交通", 4, "地铁")

print(records)
print("总支出：", total_amount())
```
