# 面向对象代码实例

## 示例 1：用户类

```python
class User:
    def __init__(self, name: str, age: int):
        self.name = name
        self.age = age

    def introduce(self) -> str:
        return f"我是 {self.name}，今年 {self.age} 岁"


user = User("Alice", 18)
print(user.introduce())
```

## 示例 2：银行账户

```python
class BankAccount:
    def __init__(self, owner: str):
        self.owner = owner
        self.balance = 0

    def deposit(self, amount: float):
        if amount <= 0:
            raise ValueError("存款金额必须大于 0")
        self.balance += amount

    def withdraw(self, amount: float):
        if amount > self.balance:
            raise ValueError("余额不足")
        self.balance -= amount


account = BankAccount("Alice")
account.deposit(100)
account.withdraw(30)
print(account.balance)
```

## 示例 3：图书管理

```python
class Book:
    def __init__(self, title: str, author: str):
        self.title = title
        self.author = author
        self.is_borrowed = False

    def borrow(self):
        if self.is_borrowed:
            print("已被借出")
        else:
            self.is_borrowed = True
            print("借阅成功")


book = Book("Python 入门", "匿名作者")
book.borrow()
```

## 示例 4：继承

```python
class Animal:
    def speak(self):
        return "..."


class Cat(Animal):
    def speak(self):
        return "喵"


class Dog(Animal):
    def speak(self):
        return "汪"


animals = [Cat(), Dog()]

for animal in animals:
    print(animal.speak())
```

## 示例 5：数据类

```python
from dataclasses import dataclass


@dataclass
class Product:
    name: str
    price: float
    stock: int


product = Product("键盘", 199.0, 10)
print(product)
```
