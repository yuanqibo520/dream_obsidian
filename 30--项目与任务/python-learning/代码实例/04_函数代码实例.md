# 函数代码实例

## 示例 1：计算 BMI

```python
def calculate_bmi(weight: float, height: float) -> float:
    return weight / height ** 2


bmi = calculate_bmi(70, 1.75)
print(f"BMI: {bmi:.2f}")
```

## 示例 2：判断质数

```python
def is_prime(number: int) -> bool:
    if number < 2:
        return False

    for i in range(2, int(number ** 0.5) + 1):
        if number % i == 0:
            return False

    return True


print(is_prime(17))
```

## 示例 3：计算列表平均值

```python
def average(numbers: list[float]) -> float:
    if not numbers:
        return 0
    return sum(numbers) / len(numbers)


print(average([80, 90, 100]))
```

## 示例 4：默认参数

```python
def create_user(name: str, role: str = "普通用户") -> dict:
    return {"name": name, "role": role}


print(create_user("Alice"))
print(create_user("Bob", "管理员"))
```

## 示例 5：可变参数

```python
def add_all(*numbers: int) -> int:
    total = 0
    for number in numbers:
        total += number
    return total


print(add_all(1, 2, 3, 4))
```

## 示例 6：关键字参数

```python
def print_profile(**profile):
    for key, value in profile.items():
        print(key, value)


print_profile(name="Alice", age=18, city="Shanghai")
```
