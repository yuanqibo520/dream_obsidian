#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 邻接表节点结构
typedef struct AdjNode {
    int vertex;          // 邻接顶点的索引
    struct AdjNode* next;// 下一个邻接节点
} AdjNode;

// 邻接表头结构
typedef struct AdjList {
    AdjNode* head;       // 邻接表的头指针
} AdjList;

// 图结构（邻接表实现）
typedef struct Graph {
    int numVertices;     // 顶点总数
    AdjList* array;      // 邻接表数组
} Graph;

// 创建新的邻接节点
AdjNode* createAdjNode(int v) {
    AdjNode* newNode = (AdjNode*)malloc(sizeof(AdjNode));
    newNode->vertex = v;
    newNode->next = NULL;
    return newNode;
}

// 创建图（初始化邻接表）
Graph* createGraph(int vertices) {
    Graph* graph = (Graph*)malloc(sizeof(Graph));
    graph->numVertices = vertices;
    graph->array = (AdjList*)malloc(vertices * sizeof(AdjList));

    // 初始化每个邻接表的头节点为NULL
    for (int i = 0; i < vertices; i++) {
        graph->array[i].head = NULL;
    }
    return graph;
}

// 添加有向边：from -> to
void addEdge(Graph* graph, int from, int to) {
    // 创建新节点并插入到from的邻接表头部
    AdjNode* newNode = createAdjNode(to);
    newNode->next = graph->array[from].head;
    graph->array[from].head = newNode;
}

// 拓扑排序（Kahn算法）
void topologicalSort(Graph* graph) {
    int numVertices = graph->numVertices;
    int* inDegree = (int*)malloc(numVertices * sizeof(int));
    int* topoOrder = (int*)malloc(numVertices * sizeof(int)); // 存储拓扑序列
    int topoIndex = 0; // 拓扑序列的索引
    int queue[numVertices]; // 数组模拟队列，存储入度为0的顶点
    int front = 0, rear = 0; // 队列的头尾指针

    // 初始化所有顶点的入度为0
    memset(inDegree, 0, numVertices * sizeof(int));

    // 计算每个顶点的入度
    for (int i = 0; i < numVertices; i++) {
        AdjNode* temp = graph->array[i].head;
        while (temp != NULL) {
            inDegree[temp->vertex]++;
            temp = temp->next;
        }
    }

    // 将所有入度为0的顶点加入队列
    for (int i = 0; i < numVertices; i++) {
        if (inDegree[i] == 0) {
            queue[rear++] = i;
        }
    }

    // 处理队列中的顶点
    while (front < rear) {
        // 取出队列头部顶点
        int current = queue[front++];
        topoOrder[topoIndex++] = current; // 加入拓扑序列

        // 遍历当前顶点的所有邻接顶点，入度减1
        AdjNode* temp = graph->array[current].head;
        while (temp != NULL) {
            int adjVertex = temp->vertex;
            inDegree[adjVertex]--;
            // 若邻接顶点入度变为0，加入队列
            if (inDegree[adjVertex] == 0) {
                queue[rear++] = adjVertex;
            }
            temp = temp->next;
        }
    }

    // 校验：若拓扑序列长度小于顶点数，说明有环
    if (topoIndex != numVertices) {
        printf("图中存在环，无法进行拓扑排序！\n");
        free(inDegree);
        free(topoOrder);
        return;
    }

    // 输出拓扑序列
    printf("拓扑排序结果：");
    for (int i = 0; i < topoIndex; i++) {
        printf("%d ", topoOrder[i]);
    }
    printf("\n");

    // 释放内存
    free(inDegree);
    free(topoOrder);
}

// 释放图的内存
void freeGraph(Graph* graph) {
    if (graph == NULL) return;
    for (int i = 0; i < graph->numVertices; i++) {
        AdjNode* temp = graph->array[i].head;
        while (temp != NULL) {
            AdjNode* next = temp->next;
            free(temp);
            temp = next;
        }
    }
    free(graph->array);
    free(graph);
}

// 测试示例
int main() {
    // 示例图：6个顶点，边为 5->2,5->0,4->0,4->1,2->3,3->1
    // 拓扑排序结果（可能有多种）：4 5 2 0 3 1  或 5 4 2 3 0 1 等
    int vertices = 6;
    Graph* graph = createGraph(vertices);
    addEdge(graph, 5, 2);
    addEdge(graph, 5, 0);
    addEdge(graph, 4, 0);
    addEdge(graph, 4, 1);
    addEdge(graph, 2, 3);
    addEdge(graph, 3, 1);
	
    topologicalSort(graph);

    // 测试含环的图（取消注释查看效果）
    // addEdge(graph, 1, 5); // 新增边1->5，形成环，输出"图中存在环"
    // topologicalSort(graph);

    freeGraph(graph);
    return 0;
}
