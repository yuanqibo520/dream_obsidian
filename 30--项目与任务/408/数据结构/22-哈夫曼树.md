#include <stdio.h>
#include <stdlib.h>

// 哈夫曼树节点结构
typedef struct HuffmanNode {
    int weight;             // 权值（叶子节点为字符频率，非叶子为合并后的和）
    struct HuffmanNode *left, *right; // 左右子树
} HuffmanNode;

// 最小堆结构（用于快速获取最小权值节点）
typedef struct MinHeap {
    int size;               // 当前堆大小
    int capacity;           // 堆容量
    HuffmanNode** array;    // 存储节点指针的数组
} MinHeap;

// 创建单个哈夫曼节点
HuffmanNode* createNode(int weight) {
    HuffmanNode* node = (HuffmanNode*)malloc(sizeof(HuffmanNode));
    node->weight = weight;
    node->left = node->right = NULL;
    return node;
}

// 创建最小堆
MinHeap* createMinHeap(int capacity) {
    MinHeap* heap = (MinHeap*)malloc(sizeof(MinHeap));
    heap->size = 0;
    heap->capacity = capacity;
    heap->array = (HuffmanNode**)malloc(capacity * sizeof(HuffmanNode*));
    return heap;
}

// 交换堆中两个节点
void swapNodes(HuffmanNode** a, HuffmanNode** b) {
    HuffmanNode* temp = *a;
    *a = *b;
    *b = temp;
}

// 最小堆调整（维护堆的性质：父节点权值 < 子节点）
void minHeapify(MinHeap* heap, int idx) {
    int smallest = idx;
    int left = 2 * idx + 1;  // 左孩子索引
    int right = 2 * idx + 2; // 右孩子索引

    // 找到权值最小的节点
    if (left < heap->size && heap->array[left]->weight < heap->array[smallest]->weight) {
        smallest = left;
    }
    if (right < heap->size && heap->array[right]->weight < heap->array[smallest]->weight) {
        smallest = right;
    }

    // 若最小节点不是当前节点，交换并递归调整
    if (smallest != idx) {
        swapNodes(&heap->array[idx], &heap->array[smallest]);
        minHeapify(heap, smallest);
    }
}

// 提取堆中权值最小的节点
HuffmanNode* extractMin(MinHeap* heap) {
    HuffmanNode* minNode = heap->array[0];
    heap->array[0] = heap->array[heap->size - 1]; // 最后一个节点移到堆顶
    heap->size--;
    minHeapify(heap, 0); // 重新调整堆
    return minNode;
}

// 插入节点到最小堆
void insertMinHeap(MinHeap* heap, HuffmanNode* node) {
    heap->size++;
    int i = heap->size - 1;

    // 向上调整：父节点权值更大则交换
    while (i > 0 && node->weight < heap->array[(i - 1) / 2]->weight) {
        heap->array[i] = heap->array[(i - 1) / 2];
        i = (i - 1) / 2;
    }
    heap->array[i] = node;
}

// 构建最小堆
void buildMinHeap(MinHeap* heap) {
    int n = heap->size - 1;
    // 从最后一个非叶子节点开始调整
    for (int i = (n - 1) / 2; i >= 0; i--) {
        minHeapify(heap, i);
    }
}

// 构建哈夫曼树
HuffmanNode* buildHuffmanTree(int* weights, int n) {
    HuffmanNode *left, *right, *parent;

    // 1. 初始化最小堆：每个权值对应一个叶子节点
    MinHeap* heap = createMinHeap(n);
    for (int i = 0; i < n; i++) {
        heap->array[i] = createNode(weights[i]);
    }
    heap->size = n;
    buildMinHeap(heap);

    // 2. 循环合并最小节点，直到堆中只剩一个节点
    while (heap->size > 1) {
        // 取出两个最小权值节点
        left = extractMin(heap);
        right = extractMin(heap);

        // 合并为新节点（权值=两节点之和，用-1标识非叶子节点）
        parent = createNode(left->weight + right->weight);
        parent->left = left;
        parent->right = right;

        // 新节点入堆
        insertMinHeap(heap, parent);
    }

    // 3. 最后一个节点是哈夫曼树根节点
    HuffmanNode* root = extractMin(heap);
    free(heap->array);
    free(heap);
    return root;
}

// 释放哈夫曼树内存
void freeHuffmanTree(HuffmanNode* root) {
    if (root == NULL) return;
    freeHuffmanTree(root->left);
    freeHuffmanTree(root->right);
    free(root);
}

// 测试示例
int main() {
    // 示例：叶子节点权值数组
    int weights[] = {5, 9, 12, 13, 16, 45};
    int n = sizeof(weights) / sizeof(weights[0]);

    // 1. 构建哈夫曼树
    HuffmanNode* root = buildHuffmanTree(weights, n);


    // 4. 释放内存
    freeHuffmanTree(root);

    return 0;
}
