#include <stdio.h> 

// 二分查找：在已排序区间 [left, right] 中找到待插入元素的位置 
int binarySearch(int arr[], int left, int right, int target)
 { 
	 while (left <= right) 
	 { // 计算中间位置（避免 left+right 溢出） 
		int mid = left + (right - left) / 2; 
		if (arr[mid] > target) 
		{ // 目标值更小，插入位置在左半区 
			 right = mid - 1; 
		} 
		else 
		{// 目标值更大/相等，插入位置在右半区 
			left = mid + 1; 
		}
	} // 最终left就是插入位置 return left; 
} // 二分插入排序核心函数

void binaryInsertionSort(int arr[], int n) 
{ // 从第二个元素开始（第一个元素默认已排序） 
	for (int i = 1; i < n; i++) 
	{ // 保存当前待插入的元素 
		int current = arr[i]; // 已排序区间的右边界 
		int j = i - 1; // 用二分查找找到插入位置 
		int insertPos = binarySearch(arr, 0, j, current); 
		// 将插入位置后的元素整体后移一位 
		while (j >= insertPos)
		{
			arr[j + 1] = arr[j]; 
			j--;
		} 
		// 插入当前元素到正确位置
			arr[insertPos] = current;
	} 
} // 辅助函数：打印数组 

void printArray(int arr[], int n) 
{
	for (int i = 0; i < n; i++)
	{ 
		printf("%d ", arr[i]); 
	} 
		printf("\n"); } // 主函数测试 

int main() { // 测试数组 
	int arr[] = {7, 3, 5, 9, 1, 8, 2}; // 计算数组长度 
	int n = sizeof(arr) / sizeof(arr[0]); 
	printf("原数组："); 
	printArray(arr, n); // 执行二分插入排序  
	binaryInsertionSort(arr, n); 
	printf("排序后："); 
	printArray(arr, n); 
	return 0; 
}
