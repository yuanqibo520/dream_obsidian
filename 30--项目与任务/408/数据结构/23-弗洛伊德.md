```C
#include <stdio.h>
#include <limits.h>

// 常量定义
#define INF INT_MAX // 表示无穷大（不可达）
#define V 4        // 图的顶点数（示例用4个顶点）

// 打印最短距离矩阵
void printDistanceMatrix(int dist[V][V]) {
    printf("顶点间最短距离矩阵（INF表示不可达）：\n");
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            if (dist[i][j] == INF) {
                printf("INF\t");
            } else {
                printf("%d\t", dist[i][j]);
            }
        }
        printf("\n");
    }
}

// 还原最短路径（辅助函数）
void printPath(int path[V][V], int i, int j) {
    if (path[i][j] == -1) { // 无中转点，直接到达
        if (i != j) {
            printf(" → %d", j);
        }
        return;
    }
    // 递归打印 i → path[i][j] 的路径
    printPath(path, i, path[i][j]);
    // 打印 path[i][j] → j 的路径
    printPath(path, path[i][j], j);
}

// 打印所有顶点对的最短路径
void printAllPaths(int dist[V][V], int path[V][V]) {
    printf("\n所有顶点对的最短路径：\n");
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            if (i != j) {
                printf("从 %d 到 %d：", i, j);
                if (dist[i][j] == INF) {
                    printf("无路径\n");
                } else {
                    printf("%d", i);
                    printPath(path, i, j);
                    printf("  距离：%d\n", dist[i][j]);
                }
            }
        }
    }
}

// Floyd算法核心实现（含路径记录）
void floydWarshall(int graph[V][V]) {
    int dist[V][V]; // 存储最终最短距离
    int path[V][V]; // 存储路径：path[i][j]表示i→j的中转点

    // 1. 初始化距离矩阵和路径矩阵
    for (int i = 0; i < V; i++) 
    {
        for (int j = 0; j < V; j++) 
        {
            dist[i][j] = graph[i][j];
            // 初始化路径：无中转点则为-1，自身到自身也为-1
            if (i == j || graph[i][j] == INF) {
                path[i][j] = -1;
            } else {
                path[i][j] = -1; // 初始无中转点
            }
        }
    }

    // 2. 核心三层循环：k为中转点，i为起点，j为终点
    for (int k = 0; k < V; k++) { // 遍历所有中转点
        for (int i = 0; i < V; i++) { // 遍历所有起点
            for (int j = 0; j < V; j++) { // 遍历所有终点
                // 跳过无穷大的情况（避免溢出）
                if (dist[i][k] != INF && dist[k][j] != INF) {
                    // 中转路径更短，则更新距离和路径
                    if (dist[i][j] > dist[i][k] + dist[k][j]) {
                        dist[i][j] = dist[i][k] + dist[k][j];
                        path[i][j] = k; // 记录中转点k
                    }
                }
            }
        }
    }

    // 3. 输出结果
    printDistanceMatrix(dist);
    printAllPaths(dist, path);
}

// 测试示例
int main() {
    // 示例图的邻接矩阵（4个顶点）
    // 解释：graph[i][j]表示i到j的直接边权，INF表示无直接边
    int graph[V][V] = {
        {0,   5,   INF, 10},
        {INF, 0,   3,   INF},
        {INF, INF, 0,   1},
        {INF, INF, INF, 0}
    };

    // 执行Floyd算法
    floydWarshall(graph);

    return 0;
}
```
