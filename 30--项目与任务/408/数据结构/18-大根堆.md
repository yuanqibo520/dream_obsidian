void BuildMaxHeap (int A[],int len)
{
	for(int i =len/2;i>0;i--)
	HeadAdjust(A,i,len);
}
void HeadAdjust(int A[],int k, int len)
{
	for(int i = k*2;i<=len;i*=2)
	{
		if(i<len&&A[i]<A[i++])
			i++;
		if(A[k]>A[i])break;
		else
		{
			A[k]=A[i];
			k=i;
		}
	}
	A[k]=A[0];
}
