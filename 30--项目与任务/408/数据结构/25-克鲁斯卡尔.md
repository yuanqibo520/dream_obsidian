#include <stdio.h>
#include <stdlib.h>

// 边的结构定义
typedef struct Edge {
    int src;    // 起点
    int dest;   // 终点
    int weight; // 权值
} Edge;

// 图的结构定义（存储所有边）
typedef struct Graph {
    int V; // 顶点数
    int E; // 边数
    Edge* edge; // 边数组
} Graph;

// 创建图
Graph* createGraph(int V, int E) {
    Graph* graph = (Graph*)malloc(sizeof(Graph));
    graph->V = V;
    graph->E = E;
    graph->edge = (Edge*)malloc(E * sizeof(Edge));
    return graph;
}

// 排序用的比较函数（按边权升序）
int compare(const void* a, const void* b) {
    Edge* edgeA = (Edge*)a;
    Edge* edgeB = (Edge*)b;
    return edgeA->weight - edgeB->weight;
}

// 并查集：查找顶点的根节点（带路径压缩）
int find(int parent[], int i) {
    if (parent[i] == i) {
        return i;
    }
    // 路径压缩：将当前节点直接指向根节点，加速后续查找
    parent[i] = find(parent, parent[i]);
    return parent[i];
}

// 并查集：合并两个连通分量（按秩合并）
void Union(int parent[], int rank[], int x, int y) {
    int xRoot = find(parent, x);
    int yRoot = find(parent, y);

    // 按秩合并：秩小的树挂到秩大的树根下
    if (rank[xRoot] < rank[yRoot]) {
        parent[xRoot] = yRoot;
    } else if (rank[xRoot] > rank[yRoot]) {
        parent[yRoot] = xRoot;
    } else {
        // 秩相等，任选一个作为根，秩+1
        parent[yRoot] = xRoot;
        rank[xRoot]++;
    }
}

// 克鲁斯卡尔算法核心实现
void kruskalMST(Graph* graph) {
    int V = graph->V;
    Edge result[V]; // 存储MST的边（MST最多V-1条边）
    int e = 0;      // 结果数组的索引
    int i = 0;      // 排序后边的遍历索引

    // 1. 将所有边按权值升序排序
    qsort(graph->edge, graph->E, sizeof(graph->edge[0]), compare);

    // 2. 初始化并查集
    int* parent = (int*)malloc(V * sizeof(int));
    int* rank = (int*)malloc(V * sizeof(int));
    for (int v = 0; v < V; v++) {
        parent[v] = v; // 初始时每个顶点的父节点是自己
        rank[v] = 0;   // 初始秩为0
    }

    // 3. 遍历排序后的边，选边构建MST
    while (e < V - 1 && i < graph->E) {
        Edge currEdge = graph->edge[i++]; // 取当前权值最小的边

        // 查找两个顶点的根节点
        int x = find(parent, currEdge.src);
        int y = find(parent, currEdge.dest);

        // 若根节点不同（加入后不形成环），则加入MST
        if (x != y) {
            result[e++] = currEdge;
            Union(parent, rank, x, y);
        }
        // 否则跳过这条边（形成环）
    }

    // 4. 输出MST结果
    printf("最小生成树（Kruskal）的边：\n");
    printf("起点 → 终点 \t 权值\n");
    int totalWeight = 0; // MST的总权值
    for (i = 0; i < e; i++) {
        printf("%d    →   %d \t %d\n", result[i].src, result[i].dest, result[i].weight);
        totalWeight += result[i].weight;
    }
    printf("最小生成树总权值：%d\n", totalWeight);

    // 释放内存
    free(parent);
    free(rank);
}

// 测试示例
int main() {
    /* 示例图（5个顶点，7条边）：
       顶点：0,1,2,3,4
       边：
       0-1 (2)  0-3 (6)  1-2 (3)  1-3 (8)  1-4 (5)  2-4 (7)  3-4 (9)
    */
    int V = 5; // 顶点数
    int E = 7; // 边数
    Graph* graph = createGraph(V, E);

    // 初始化所有边
    graph->edge[0] = (Edge){0, 1, 2};
    graph->edge[1] = (Edge){0, 3, 6};
    graph->edge[2] = (Edge){1, 2, 3};
    graph->edge[3] = (Edge){1, 3, 8};
    graph->edge[4] = (Edge){1, 4, 5};
    graph->edge[5] = (Edge){2, 4, 7};
    graph->edge[6] = (Edge){3, 4, 9};

    // 执行克鲁斯卡尔算法
    kruskalMST(graph);

    // 释放图内存
    free(graph->edge);
    free(graph);

    return 0;
}
