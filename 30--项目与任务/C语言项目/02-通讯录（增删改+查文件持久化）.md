// 1. 定义常量和结构体
#define MAX_CONTACTS 100  // 最大联系人数量

// 提供跨平台的线程不安全/安全 strtok 替代宏：MSVC 使用 strtok_s，其他平台使用 strtok_r
#if defined(_MSC_VER)
    #define SAFE_STRTOK(str, delim, saveptr) strtok_s(str, delim, saveptr)
#else
    #define SAFE_STRTOK(str, delim, saveptr) strtok_r(str, delim, saveptr)
#endif

// 4. 函数实现
// helper 实现
static void flushInput(void)
{
    int ch;
    // 读取并丢弃直到遇到换行或 EOF
    while ((ch = getchar()) != '\n' && ch != EOF) {}
}

static FILE* safeFopen(const char* filename, const char* mode)
{
#ifdef _MSC_VER
    FILE* fp = NULL;
    if (fopen_s(&fp, filename, mode) == 0) {
        return fp;
    }
    return NULL;
#else
    return fopen(filename, mode);
#endif
}

static void safeStrCopy(char* dest, size_t destSize, const char* src)
{
#ifdef _MSC_VER
    strcpy_s(dest, destSize, src);
#else
    // strncpy 保证不会溢出，但可能不以 '\0' 结尾，故需手动设置最后一位
    strncpy(dest, src, destSize - 1);
    dest[destSize - 1] = '\0';
#endif
}

// 初始化通讯录（清空数量）
void initContactBook(ContactBook* book) {
    book->count = 0;
    memset(book->contacts, 0, sizeof(book->contacts));
}

// 从CSV文件加载数据（使用 fgets + SAFE_STRTOK 避免 C4996 警告）
void loadFromFile(ContactBook* book) {
    FILE* fp = safeFopen(FILE_NAME, "r");
    if (fp == NULL) {
        printf("未找到历史数据文件，将创建新文件！\n");
        return;
    }

    char line[512];
    Contact temp;
    while (fgets(line, sizeof(line), fp) != NULL) {
        // 去除行末换行符
        size_t len = strlen(line);
        if (len > 0 && (line[len - 1] == '\n' || line[len - 1] == '\r')) {
            line[len - 1] = '\0';
            len--;
            if (len > 0 && line[len - 1] == '\r') { // 处理 \r\n
                line[len - 1] = '\0';
            }
        }

        // 使用 SAFE_STRTOK 分割 CSV（简单实现，不处理字段内逗号或引号）
        char* saveptr = NULL;
        char* token = SAFE_STRTOK(line, ",", &saveptr);
        if (token == NULL) continue;
        token = SAFE_STRTOK(NULL, ",", &saveptr);
        safeStrCopy(temp.phone, MAX_PHONE, token ? token : "");

        token = SAFE_STRTOK(NULL, ",", &saveptr);
        safeStrCopy(temp.email, MAX_EMAIL, token ? token : "");

        token = SAFE_STRTOK(NULL, ",", &saveptr); // 剩下的全部视为地址（可包含逗号）
        safeStrCopy(temp.addr, MAX_ADDR, token ? token : "");

        // 检查数组是否已满
        if (book->count >= MAX_CONTACTS) {
            printf("历史数据过多，仅加载前%d条！\n", MAX_CONTACTS);
            break;
        }
        // 复制到通讯录数组
        memcpy(&book->contacts[book->count], &temp, sizeof(Contact));
        book->count++;
    }

    fclose(fp);
    printf("成功加载%d条联系人数据！\n", book->count);
}

// 保存数据到CSV文件
void saveToFile(ContactBook* book) {
    FILE* fp = safeFopen(FILE_NAME, "w");
    if (fp == NULL) {
        printf("文件打开失败，数据保存失败！\n");
        return;
    }

    // 遍历数组，按CSV格式写入
    for (int i = 0; i < book->count; i++) {
        fprintf(fp, "%s,%s,%s,%s\n",
            book->contacts[i].name,
            book->contacts[i].phone,
            book->contacts[i].email,
            book->contacts[i].addr);
    }

    fclose(fp);
    printf("成功保存%d条联系人数据到文件！\n", book->count);
}

// 添加联系人
void addContact(ContactBook* book) {
    if (book->count >= MAX_CONTACTS) {
        printf("通讯录已满，无法添加新联系人！\n");
        return;
    }

    char name[MAX_NAME];
    printf("请输入联系人姓名：");
    // 这里 main 已经在读取选项后 flush 过 stdin，直接使用 fgets
    if (fgets(name, MAX_NAME, stdin) == NULL) return;
    // 去除fgets读取的换行符
    name[strcspn(name, "\n")] = '\0';

    // 检查姓名是否重复
    for (int i = 0; i < book->count; i++) {
        if (strcmp(book->contacts[i].name, name) == 0) {
            printf("姓名重复，无法添加！\n");
            return;
        }
    }

    // 填充联系人信息
    Contact* newContact = &book->contacts[book->count];
    safeStrCopy(newContact->name, MAX_NAME, name);

    printf("请输入电话号码：");
    if (fgets(newContact->phone, MAX_PHONE, stdin) == NULL) return;
    newContact->phone[strcspn(newContact->phone, "\n")] = '\0';

    printf("请输入邮箱：");
    if (fgets(newContact->email, MAX_EMAIL, stdin) == NULL) return;
    newContact->email[strcspn(newContact->email, "\n")] = '\0';

    printf("请输入地址：");
    if (fgets(newContact->addr, MAX_ADDR, stdin) == NULL) return;
    newContact->addr[strcspn(newContact->addr, "\n")] = '\0';

    book->count++;
    printf("联系人添加成功！\n");
}

// 删除联系人（按姓名）
void deleteContact(ContactBook* book) {
    if (book->count == 0) {
        printf("通讯录为空，无联系人可删除！\n");
        return;
    }

    char name[MAX_NAME];
    printf("请输入要删除的联系人姓名：");
    if (fgets(name, MAX_NAME, stdin) == NULL) return;
    name[strcspn(name, "\n")] = '\0';

    // 查找联系人
    int index = -1;
    for (int i = 0; i < book->count; i++) {
        if (strcmp(book->contacts[i].name, name) == 0) {
            index = i;
            break;
        }
    }

    if (index == -1) {
        printf("未找到该联系人！\n");
        return;
    }

    // 移除元素：后面的元素向前移动
    for (int i = index; i < book->count - 1; i++) {
        memcpy(&book->contacts[i], &book->contacts[i + 1], sizeof(Contact));
    }
    book->count--;
    printf("联系人删除成功！\n");
}

    }

    // 修改信息（保留原信息：直接回车则不修改）
    Contact* contact = &book->contacts[index];
    char temp[MAX_ADDR];

    printf("当前姓名：%s（不可修改）\n", contact->name);

    printf("请输入新电话号码（回车保留原号码：%s）：", contact->phone);
    if (fgets(temp, MAX_PHONE, stdin) != NULL) {
        temp[strcspn(temp, "\n")] = '\0';
        if (strlen(temp) > 0) {
            safeStrCopy(contact->phone, MAX_PHONE, temp);
        }
    }

    printf("请输入新邮箱（回车保留原邮箱：%s）：", contact->email);
    if (fgets(temp, MAX_EMAIL, stdin) != NULL) {
        temp[strcspn(temp, "\n")] = '\0';
        if (strlen(temp) > 0) {
            safeStrCopy(contact->email, MAX_EMAIL, temp);
        }
    }

    printf("请输入新地址（回车保留原地址：%s）：", contact->addr);
    if (fgets(temp, MAX_ADDR, stdin) != NULL) {
        temp[strcspn(temp, "\n")] = '\0';
        if (strlen(temp) > 0) {
            safeStrCopy(contact->addr, MAX_ADDR, temp);
        }
    }

    printf("联系人信息修改成功！\n");
}

// 搜索联系人（姓名/电话模糊匹配）
void searchContact(ContactBook* book) {
    if (book->count == 0) {
        printf("通讯录为空，无联系人可搜索！\n");
        return;
    }

    char keyword[MAX_NAME];
    printf("请输入搜索关键词（姓名/电话）：");
    if (fgets(keyword, MAX_NAME, stdin) == NULL) return;
    keyword[strcspn(keyword, "\n")] = '\0';

    int found = 0;
    printf("\n===== 搜索结果 =====\n");
    for (int i = 0; i < book->count; i++) {
        Contact* c = &book->contacts[i];
        // 模糊匹配：关键词是姓名/电话的子串
        if ((keyword[0] != '\0') &&
            (strstr(c->name, keyword) != NULL || strstr(c->phone, keyword) != NULL)) {
            printf("姓名: %s\n", c->name);
            printf("电话: %s\n", c->phone);
            printf("邮箱: %s\n", c->email);
            printf("地址: %s\n", c->addr);
            printf("-------------------\n");
            found++;
        }
    }

    if (found == 0) {
        printf("未找到匹配的联系人！\n");
    }
    else {
        printf("共找到%d条匹配结果！\n", found);
    }
}

// 按姓名排序（冒泡排序）
void sortContacts(ContactBook* book) {
    if (book->count <= 1) {
        return;  // 无需排序
    }

    for (int i = 0; i < book->count - 1; i++) {
        for (int j = 0; j < book->count - 1 - i; j++) {
            // 按字符串字典序比较
            if (strcmp(book->contacts[j].name, book->contacts[j + 1].name) > 0) {
                // 交换两个联系人
                Contact temp = book->contacts[j];
                book->contacts[j] = book->contacts[j + 1];
                book->contacts[j + 1] = temp;
            }
        }
    }
}

// 显示所有联系人（先排序再显示）
void showAllContacts(ContactBook* book) {
    if (book->count == 0) {
        printf("通讯录为空！\n");
        return;
    }

    // 先排序再显示
    sortContacts(book);
    printf("\n===== 所有联系人（按姓名排序） =====\n");
    for (int i = 0; i < book->count; i++) {
        printf("第%d条：\n", i + 1);
        printf("姓名: %s\n", book->contacts[i].name);
        printf("电话: %s\n", book->contacts[i].phone);
        printf("邮箱: %s\n", book->contacts[i].email);
        printf("地址: %s\n", book->contacts[i].addr);
        printf("-------------------\n");
    }
    printf("总计%d条联系人！\n", book->count);
}

// 导出到新文件
void exportToFile(ContactBook* book) {
    if (book->count == 0) {
        printf("通讯录为空，无数据可导出！\n");
        return;
    }

    char newFileName[256];
    printf("请输入导出文件名（如export_contacts.txt）：");
    if (fgets(newFileName, sizeof(newFileName), stdin) == NULL) return;
    newFileName[strcspn(newFileName, "\n")] = '\0';

    FILE* fp = safeFopen(newFileName, "w");
    if (fp == NULL) {
        printf("导出文件创建失败！\n");
        return;
    }

    // 先排序再导出
    sortContacts(book);
    for (int i = 0; i < book->count; i++) {
        fprintf(fp, "%s,%s,%s,%s\n",
            book->contacts[i].name,
            book->contacts[i].phone,
            book->contacts[i].email,
            book->contacts[i].addr);
    }

    fclose(fp);
    printf("成功导出%d条联系人数据到%s！\n", book->count, newFileName);
}
