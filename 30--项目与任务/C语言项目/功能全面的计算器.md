//数字栈和符号栈定义
typedef struct {
	double *num;
	int top;
}StackNum;
typedef struct {
	char *op;
	int top;
}StackOp;
//栈初始化函数
void InitStackNum(StackNum *N,int len)
{
	N->top = -1;
	N->num = (double*)malloc(sizeof(int) * (len) / 2);
}
void InitStackOp(StackOp* O,int len)
{
	O->top = -1;
	O->op = (char*)malloc(sizeof(int) * (len) / 2);
}
//字符判断函数
bool is_space(const char n)
{
	if (n == ' ')
		return true;
	return false;
}
bool is_digit(const char n)
{
	return n >= '0' && n <= '9';
}
//入栈函数
bool PushNum(StackNum* N, const double ret)
{
	N->num[++N->top] = ret;
	return true;
}
bool PushOp(StackOp* O, const char n)
{
	O->op[++O->top] = n;
	return true;
}
//取栈顶元素函数
char PeekOp(StackOp* O)
{
	return O->op[O->top];
}
//出栈函数
double PopNum(StackNum* N)
{
	return N->num[N->top--];
}
char PopOp(StackOp* O)
{
	return O->op[O->top--];
}
//栈空判断函数
int isEmptyOp(StackOp* O)
{
	return O->top == -1;
}
int isEmptyNum(StackNum* O)
{
	return O->top == -1;
}
//符号判断函数
bool is_symbol(const char a)
{
	return (a == '+' || a == '-' || a == '*' || a == '/' || a == '^');
}
//符号优先级函数
int Precedence(const char c)
{
	if (c == '+' || c == '-') return 1;
	if (c == '*' || c == '/') return 2;
	if (c == '^') return 3;
	return 0;
}
//计算函数
double applyOp(double a, double b, char op ,int * flag_div_zero)
{
	switch (op) {
	case '+':return a + b;
		break;
	case '-':return a - b;
		break;
	case '*':return a * b;
		break;
	case '/': {
		if (b != 0)
			return a / b;
		if (b == 0)
		*flag_div_zero = -1;
		return 0; // 这里返回0，后续检测到除以0时会报错
	}
			break;
	case '^': {
		return pow(a, b);
	}
	default: Error_Input();// 检测 x/0 和 符号正确，错误显示指明原因重新输入
	}
}

//核心计算函数
int Core_Function(const char arr[],double *num,int len)
{
	//栈初始化
	StackNum N;
	StackOp O;
	InitStackNum(&N,len);
	InitStackOp(&O,len);
	for (int i = 0; arr[i] != '\0'; i++)
	{
		//空格跳过
		if (is_space(arr[i]))
			continue; 
		//左括号入栈
		if (arr[i] == '(')
		{
			PushOp(&O, arr[i]);
			continue;
		}
		//负号处理原理：-x 等价于 0 - x
		if (arr[i] == '-' &&
			(i == 0 || arr[i - 1] == '('))
		{
			PushNum(&N, 0);
		}
		//数字入栈
		if (is_digit(arr[i]) || arr[i] == '.')
		{
			double ret = 0.0;
			if (is_digit(arr[i]))//整数部分处理
			{
				while (is_digit(arr[i]))
				{
					ret = ret * 10 + (arr[i] - '0');
					i++;
				}
			}
			if (arr[i] == '.')//小数部分处理
			{
				i++;
				int j = 1;
				double Decimal_part = 1;
				while (is_digit(arr[i]))
				{
					Decimal_part /= pow(10, j++);
					ret += Decimal_part * (arr[i++] - '0');
				}
			}
			i--;//指针回退
			PushNum(&N, ret);
		}
		//右括号出栈
		if (arr[i] == ')')
		{
			while (PeekOp(&O) != '(' && !isEmptyOp(&O))
			{
				int flag_div_zero = 0;
				char op = PopOp(&O);
				double a = PopNum(&N);
				double b = PopNum(&N);
				PushNum(&N, applyOp(b, a, op, &flag_div_zero));
				if (flag_div_zero == -1)return -1;
			}
			PopOp(&O);
		}
		//符号处理
		if(!is_digit(arr[i])&&!is_symbol(arr[i]))
			return 1;//数字刚处理完，又被当成非法字符
		if (is_symbol(arr[i]))
		{
			while (PeekOp(&O) != '(' && !isEmptyOp(&O) && Precedence(PeekOp(&O)) >= Precedence(arr[i]))
			{
				int flag_div_zero = 0;
				char op = PopOp(&O);
				double a = PopNum(&N);
				double b = PopNum(&N);
				PushNum(&N, applyOp(b, a, op, &flag_div_zero));
				if (flag_div_zero == -1)return -1;
			}
			PushOp(&O, arr[i]);
		}
		/*if (!is_symbol(arr[i]))
			Error_Input = -1;//数字刚处理完，又被当成非法字符*/
	}
	//剩余符号处理
	while (!isEmptyOp(&O))
	{
		int flag_div_zero = 0;
		char op = PopOp(&O);
		double a = PopNum(&N);
		double b = PopNum(&N);
		PushNum(&N, applyOp(b, a, op, &flag_div_zero));
		if (flag_div_zero == -1)return -1;
	}
	*num = PopNum(&N);
	//错误检测返回值
	return 0;
}
//	主函数
int main()
{
	Application_Page();
	char arr[100] = { 0 };
	fgets(arr, sizeof(arr), stdin);
	arr[strcspn(arr, "\n")] = '\0';
	int len = strlen(arr);
	while (1)
	{
		//退出判断
		if (!strcmp(arr, "exit"))
		{
			End_screen();
			break;
		}
		double ret = 0.0;
		int end = Core_Function(arr,&ret,len);
		//错误检测输出界面
		if (end == -1)Error_Logic();
		if (end == 0)repetitive_scene(ret);
		if (end == 1)Error_Input();
		//获取下一行输入
		fgets(arr, sizeof(arr), stdin);
		arr[strcspn(arr, "\n")] = '\0';
	}
	return 0;
}
