#include<stdio.h>
#include<stdlib.h>
#include<string.h>
// 定义链表节点结构
typedef struct Phone_book {
	char name[20];
	char phone_number[20];
	char email[20];
	char address[20];
	struct Phone_book* next;
}*PbNode, PbList;

// 显示主界面
void Application_Page()
{
	printf("+==========================================+\n");
	printf("+=                                        =+\n");
	printf("+=             Mini__Contacts             =+\n");
	printf("+=                                        =+\n");
	printf("+==========================================+\n");
	printf("+=        <Function__Introduction>        =+\n");
	printf("+=----------------------------------------=+\n");
	printf("+=1.Add               2.Delete            =+\n");
	printf("+=3.Edit              4.Query             =+\n");
	printf("+=5.Show All          6.Exit              =+\n");
	printf("+==========================================+\n");
	printf("+=Please enter an expression:");
}

// 初始化头节点（空链表）
PbNode InitPbNode()
{
	PbNode List = (PbNode)malloc(sizeof(PbList));
	if (List == NULL) { // 内存分配失败检查
		printf("+==========================================+\n");
		printf("+=        Memory allocation failed!       =+\n");
		printf("+==========================================+\n");
		exit(1);
	}
	memset(List, 0, sizeof(PbList)); // 初始化内存为0
	List->next = NULL;
	return List;
}

// 未找到联系人提示
void Error()
{
	printf("+==========================================+\n");
	printf("+=               Not Found!               =+\n");
	printf("+==========================================+\n");
	printf("+=Please enter an expression:");
}

// 添加联系人
void Add(PbNode head)
{
	// 找到链表最后一个节点
	PbNode tail = head;
	while (tail->next != NULL) {
		tail = tail->next;
	}

	// 创建新节点并初始化
	PbNode newNode = (PbNode)malloc(sizeof(PbList));
	if (newNode == NULL) {
		printf("+==========================================+\n");
		printf("+=        Memory allocation failed!       =+\n");
		printf("+==========================================+\n");
		return;
	}
	memset(newNode, 0, sizeof(PbList));
	newNode->next = NULL;

	// 输入联系人信息（修复scanf_s参数，添加缓冲区大小）
	printf("+==========================================+\n");
	printf("+=Please enter the name:");
	scanf_s("%s", newNode->name, (unsigned)_countof(newNode->name));
	printf("+=Please enter the phone number:");
	scanf_s("%s", newNode->phone_number, (unsigned)_countof(newNode->phone_number));
	printf("+=Please enter the email:");
	scanf_s("%s", newNode->email, (unsigned)_countof(newNode->email));
	printf("+=Please enter the address:");
	scanf_s("%s", newNode->address, (unsigned)_countof(newNode->address));

	// 挂载新节点到链表尾部
	tail->next = newNode;
	printf("+==========================================+\n");
	printf("+=               Add success!             =+\n");
	printf("+==========================================+\n");
	printf("+=Please enter an expression:");
}

// 删除联系人（修复版）
void Delete(PbNode head)
{
	char name[20];
	printf("+==========================================+\n");
	printf("+=Please enter the name to delete:");
	scanf_s("%s", name, (unsigned)_countof(name));

	PbNode pre = head;       // 前驱节点
	PbNode curr = head->next;// 当前节点（从第一个数据节点开始）

	// 遍历所有数据节点（修复：curr != NULL 而非 curr->next != NULL）
	while (curr != NULL) {
		if (strcmp(curr->name, name) == 0) {
			// 找到目标节点，修改指针+释放当前节点（修复free对象）
			pre->next = curr->next;
			free(curr); // 释放被删节点本身
			printf("+==========================================+\n");
			printf("+=               Delete success!          =+\n");
			printf("+==========================================+\n");
			printf("+=Please enter an expression:");
			return;
		}
		pre = curr;
		curr = curr->next;
	}
	// 未找到
	Error();
}

// 编辑联系人（修复版）
void Edit(PbNode head)
{
	char name[20];
	printf("+==========================================+\n");
	printf("+=Please enter the name to edit:");
	scanf_s("%s", name, (unsigned)_countof(name));

	// 从第一个数据节点开始遍历（修复遍历起点）
	PbNode curr = head->next;
	while (curr != NULL) {
		if (strcmp(curr->name, name) == 0) {
			char ch;
			printf("+==========================================+\n");

			// 修复：读取字符前清空输入缓冲区的换行符
			getchar();

			// 修改姓名
			printf("+=Do you want to change the name?(Y/N):");
			scanf_s("%c", &ch, 1); // 修复scanf_s参数（字符需指定长度1）
			if (ch == 'Y' || ch == 'y') {
				getchar(); // 清空换行
				printf("+=Please enter new name:");
				scanf_s("%s", curr->name, (unsigned)_countof(curr->name));
			}

			// 修改手机号
			getchar(); // 清空换行
			printf("+=Do you want to change the phone_number?(Y/N):");
			scanf_s("%c", &ch, 1);
			if (ch == 'Y' || ch == 'y') {
				getchar();
				printf("+=Please enter new phone number:");
				scanf_s("%s", curr->phone_number, (unsigned)_countof(curr->phone_number));
			}

			// 修改邮箱
			getchar();
			printf("+=Do you want to change the email?(Y/N):");
			scanf_s("%c", &ch, 1);
			if (ch == 'Y' || ch == 'y') {
				getchar();
				printf("+=Please enter new email:");
				scanf_s("%s", curr->email, (unsigned)_countof(curr->email));
			}

			// 修改地址
			getchar();
			printf("+=Do you want to change the address?(Y/N):");
			scanf_s("%c", &ch, 1);
			if (ch == 'Y' || ch == 'y') {
				getchar();
				printf("+=Please enter new address:");
				scanf_s("%s", curr->address, (unsigned)_countof(curr->address));
			}

			printf("+==========================================+\n");
			printf("+=Edit success! Please enter an expression:");
			return;
		}
		curr = curr->next;
	}
	// 未找到
	Error();
}

// 查询联系人（修复版）
void Query(PbNode head)
{
	char Query_content[20];
	printf("+==========================================+\n");
	printf("+=Please enter the Query content:");
	scanf_s("%s", Query_content, (unsigned)_countof(Query_content));

	// 从第一个数据节点开始遍历（修复遍历起点）
	PbNode curr = head->next;
	while (curr != NULL) {
		if (strcmp(curr->name, Query_content) == 0 ||
			strcmp(curr->phone_number, Query_content) == 0 ||
			strcmp(curr->email, Query_content) == 0 ||
			strcmp(curr->address, Query_content) == 0) {
			// 找到匹配项，打印信息
			printf("+==========================================+\n");
			printf("+=Found contact:\n");
			printf("+=Name:%s\n", curr->name);
			printf("+=Phone number:%s\n", curr->phone_number);
			printf("+=Email:%s\n", curr->email);
			printf("+=Address:%s\n", curr->address);
			printf("+==========================================+\n");
			printf("+=Please enter an expression:");
			return;
		}
		curr = curr->next;
	}
	// 未找到
	Error();
}

// 显示所有联系人（修复版）
void Show_All(PbNode head)
{
	int count = 1;
	PbNode curr = head->next; // 从第一个数据节点开始

	if (curr == NULL) { // 空链表提示
		printf("+==========================================+\n");
		printf("+=            No contacts yet!            =+\n");
		printf("+==========================================+\n");
		printf("+=Please enter an expression:");
		return;
	}

	// 遍历所有数据节点（修复：curr != NULL 而非 curr->next != NULL）
	while (curr != NULL) {
		printf("+==========================================+\n");
		printf("+=User-%03d Information:\n", count++);
		printf("+=Name:%s\n", curr->name);
		printf("+=Phone number:%s\n", curr->phone_number);
		printf("+=Email:%s\n", curr->email);
		printf("+=Address:%s\n", curr->address);
		printf("\n");
		curr = curr->next;
	}
	printf("+=Total contacts: %d\n", count - 1);
	printf("+==========================================+\n");
	printf("+=Please enter an expression:");
}

int main()
{
	Application_Page();
	PbNode head = InitPbNode(); // 头节点（不存数据）
	int count = 0;

	while (1) {
		// 读取用户输入（兼容scanf_s）
		if (scanf_s("%d", &count) != 1) {
			// 处理非数字输入
			while (getchar() != '\n'); // 清空输入缓冲区
			count = -1; // 触发default
		}

		switch (count) {
		case 1: Add(head); break;
