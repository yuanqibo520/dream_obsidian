#include<stdio.h>

int main()
{
	int num = 0;
	scanf("%d",&num);
	int count1 [num] = { 0 };
	int sum = 0;
	for(int i = 0 ;i<num; i++){
		scanf("%d",&count1[i]);
		sum += count1[i];
	}

	int count2[sum] = {0};
	int temp = 0;
	for(int i = 0; i < num ; i++){
		count2[count1[i]] = 1;
	}
	for(int i = 0;i<num;i++){
		for(int j = 0;j<sum;j++){
			if(count2[j]){
				temp = j+count1[i];
				if(temp<=sum)
				count2[temp] = 1;
				if(count1[i]-j)
				{
					temp = count1[i]-j;
					count2[temp] = 1;
				}
				else{
					if(j-count1[i])
					temp = j-count1[i];
					count2[temp] = 1;
				}
			}
		}
	}
	
	int end = 0;
	for(int i = 0;i<sum;i++){
		if(count2[i])
			end++;
	}
	
	printf("%d",end);
	return 0;
}
