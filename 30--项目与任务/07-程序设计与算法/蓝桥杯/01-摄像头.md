#include<stido.h>
#include<stdlib.h>
#include<string.h>

int entry_degree[count+1]={0};
// 邻接表节点结构
typedef struct AdjNode {
    int vertex;          // 邻接顶点的索引
    struct AdjNode* next;// 下一个邻接节点
} AdjNode;

// 邻接表头结构
typedef struct AdjList {
    AdjNode* head;       // 邻接表的头指针
} AdjList;

// 图结构（邻接表实现）
typedef struct Graph {
    int numVertices;     // 顶点总数
    AdjList* array;      // 邻接表数组
} Graph;

// 创建新的邻接节点
AdjNode* createAdjNode(int v) {
    AdjNode* newNode = (AdjNode*)malloc(sizeof(AdjNode));
    newNode->vertex = v;
    newNode->next = NULL;
    return newNode;
}

// 创建图（初始化邻接表）
Graph* createGraph(int vertices) {
    Graph* graph = (Graph*)malloc(sizeof(Graph));
    graph->numVertices = vertices;
    graph->array = (AdjList*)malloc(vertices * sizeof(AdjList));

    // 初始化每个邻接表的头节点为NULL
    for (int i = 0; i < vertices; i++) {
        graph->array[i].head = NULL;
    }
    return graph;
}

// 添加有向边：from -> to
void addEdge(Graph* graph, int from, int to) {
    // 创建新节点并插入到from的邻接表头部
    AdjNode* newNode = createAdjNode(to);
    newNode->next = graph->array[from].head;
    graph->array[from].head = newNode;
}

void topologicalSort(Graph* graph,int count){
	
	
	for(int i = 1; i <= count ;i++)
	{
		AdjNode * temp = graph->array[i].head;
		int Temp = 0;
		while(temp)
		{
			Temp = temp->vertex;
			entry_degree[Temp]++;
			temp = temp->next;
		}
	}
	
	int j = 1;
	while(j <= 5)
	{
		for(int i = 1;i <= count ;i++){
			if(entry_degree[i] == 0){
				AdjNode * temp = graph->array[i].head;
				int Temp = 0;
				while(temp)
				{
					Temp = temp->vertex;
					entry_degree[Temp]--;
					temp = temp->next;
				}
				graph->array[i].head = NULL;
				entry_degree[i] = -1;
			}
		}
		j++
	}
}

// 释放图的内存
void freeGraph(Graph* graph) {
    if (graph == NULL) return;
    for (int i = 0; i < graph->numVertices; i++) {
        AdjNode* temp = graph->array[i].head;
        while (temp != NULL) {
            AdjNode* next = temp->next;
            free(temp);
            temp = next;
        }
    }
    free(graph->array);
    free(graph);
}
