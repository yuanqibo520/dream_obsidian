// 哈希节点
typedef struct HashNode {
    int key;
    int idx;
    struct HashNode* next;
} HashNode;

#define HASH_SIZE 10007 // 随便一个大质数，远小于20亿

// 哈希函数
int hash_func(int key) {
    // 处理负数，取模后转成正数
    return (key % HASH_SIZE + HASH_SIZE) % HASH_SIZE;
}

// 插入哈希表
void hash_insert(HashNode** table, int key, int idx) {
    int pos = hash_func(key);
    HashNode* node = (HashNode*)malloc(sizeof(HashNode));
    node->key = key;
    node->idx = idx;
    node->next = table[pos];
    table[pos] = node;
}

// 查找哈希表
int hash_find(HashNode** table, int key) {
    int pos = hash_func(key);
    HashNode* p = table[pos];
    while(p) {
        if(p->key == key) 
        return p->idx;
        p = p->next;
    }
    return -1; // 没找到
}

// 释放哈希表
void hash_free(HashNode** table) {
    for(int i = 0; i < HASH_SIZE; i++) {
        HashNode* p = table[i];
        while(p) {
            HashNode* tmp = p;
            p = p->next;
            free(tmp);
        }
        table[i] = NULL;
    }
}

int* twoSum(int* nums, int numsSize, int target, int* returnSize) {
    *returnSize = 2;
    int* res = (int*)malloc(sizeof(int) * 2);

    // 初始化哈希表
    HashNode** hash_table = (HashNode**)calloc(HASH_SIZE, sizeof(HashNode*));

    for(int i = 0; i < numsSize; i++) {
        int need = target - nums[i];
        int idx = hash_find(hash_table, need);
        if(idx != -1) {
            res[0] = idx;
            res[1] = i;
            hash_free(hash_table);
            free(hash_table);
            return res;
        }
        hash_insert(hash_table, nums[i], i);
    }

    hash_free(hash_table);
    free(hash_table);
    return NULL;
}
